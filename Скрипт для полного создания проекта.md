const authMiddleware = `const jwt = require('jsonwebtoken');
const User = require('../models/User');

const authMiddleware = async (req, res, next) => {
    try {
        const token = req.headers.authorization?.split(' ')[1];
        if (!token) return res.status(401).json({ error: 'No token provided' });
        const decoded = jwt.verify(token, process.env.JWT_SECRET || 'your-secret-key');
        const user = await User.findById(decoded.userId);
        if (!user || !user.isActive) return res.status(401).json({ error: 'User not found or inactive' });
        req.userId = user._id;
        req.username = user.username;
        req.userRole = user.role;
        next();
    } catch (error) {
        return res.status(401).json({ error: 'Invalid token' });
    }
};

module.exports = authMiddleware;`;

## Как использовать:

1. **Создайте новую папку для проекта**:
```bash
mkdir cloud-storage
cd cloud-storage
```

2. **Сохраните скрипт** как `create-full-project.js` в этой папке

3. **Запустите скрипт**:
```bash
node create-full-project.js
```

4. **Установите зависимости**:
```bash
npm install
```

5. **Запустите сервер**:
```bash
npm start
```

6. **Откройте браузер**: http://localhost:3000
